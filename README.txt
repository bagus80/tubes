#Tata cara menggunakan aplikasi: 
1. import file perpus.sql ke phpmyadmin anda dengan nama database "perpus"
2. atau untuk setting manual, bisa buka folder application/config/database.php,
    - hostname = localhost
    - username = username phpmyadmin anda (jika tidak ada kosongkan)
    - password = password phpmyadmin anda (jika tidak ada kosongkan)
    - database = isikan dengan nama database yang baru anda import tadi
